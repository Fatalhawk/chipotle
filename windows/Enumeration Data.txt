enum DesktopCommand {
        BRIGHTNESS_UP, BRIGHTNESS_DOWN, VOLUME_UP, VOLUME_DOWN
}

enum WindowsCommand {
        CLOSE, OPEN
}

enum Program {
	WINDOWS, UNSUPPORTED
}

"companion_command

or

"companion_data
 [Program] [data] "

Possible messages from Android to PC:

"companion_command
 [Program] [Command] "

"companion_command" for commands
"companion_data" for data
